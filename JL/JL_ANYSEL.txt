# join_logo_scp用の実行内容スクリプト
#
# 内容：放送局とタイトル情報からJLスクリプトを選択して読み込む動作
#
#       Amatsukazeによる起動想定で、対応する環境変数が定義されている前提。
#       （対応バージョンを入れるなどで対応）
#       環境変数設定（CLI_IN_PATHとSERVICE_IDまたはSERVICE_ABBR）があれば
#       Amatsukaze関係なく使用可能。
#
# 次の順番でファイルが存在したら選択される
# （[環境変数TS]の所は、TS_IN_PATH、定義なければCLI_IN_PATHが使用される）
# 1. {環境変数TS}.trim.avs
#      trim情報を直接設定。div.txtもあれば使用
# 2. {環境変数TSから拡張子.tsを除く}.keyframe
#      trim情報またはカット位置を直接設定。div.txtもあれば使用
#      .keyframe2があれば短時間構成はカットする位置指定になる
# 3. JL_ini_選択読込.txt
#      放送局とタイトルの文字列一致から選択
#
# 直接カット位置設定時もchapter情報となる推測構成を行う処理を追加
#   use_autodet=0設定で従来の推測構成は行わない処理
#   use_autodet=1設定で推測構成を使用
#
# 設定値は起動選択(ChParamJL1.csv等)のOPTIONSに記載で変更することが可能
#

#----------------------------------------------
# 設定
#----------------------------------------------
Call "JL_ini_外部環境取得.txt"               # 外部設定
Default file_ts      ""                      # 未定義時の初期設定
Default file_div     "${file_ts}.div.txt"    # 分割位置記載の入力ファイル名
Default file_trim    "${file_ts}.trim.avs"   # Trim位置記載の入力ファイル名
Default file_keyf    "${file_ts:r}.keyframe" # keyframe入力ファイル名（カット位置情報）
Default file_keyf2   "${file_ts:r}.keyframe2" # keyframe入力ファイル名（短時間構成カット付chapter）
Default file_calltr  "JL_位置読込.txt"       # trim情報を直接設定するスクリプト
Default file_inisel  "JL_ini_選択読込.txt"   # JLスクリプトの選択読込スクリプト
Default keyf_fromcut 1                       # keyframe最初データ位置から（0=残す 1=カット）
Default keyf2_cutmax 00:07:00  # keyframe（短時間構成カット付chapter）時に指定時間までの構成はカット
Default use_autodet  1         # 直接位置指定時の構成推測（0=なし 1=あり）

#----------------------------------------------
# 動作
#----------------------------------------------
#--- JLスクリプト選択処理（最初に設定を読む。OPTIONSに-setで変数も変更可能） ---
Call "${file_inisel}"

#--- 上のCallで設定される情報 ---
Default jlsi_optstr  ""                # 追加オプション文字列
Default jlsi_jlrun   ""                # CALLスクリプト
Default jlsi_comment ""                # 選択コメント

#--- Trim直接指定あれば使用する ---
Default anysel_keyf_keepcut   0  # 1=keyframeカット時の内側カット補正無効
Default anysel_keyf2_divforce 1  # 1=(keyf2)構成時間から強制的にファイル分割
Default use_trim_direct 0
Default use_trim_keepcut 0
ReadCheck "${file_trim}" -RegOut found_trim -local
ReadCheck "${file_keyf}" -RegOut found_keyf -local
ReadCheck "${file_keyf2}" -RegOut found_keyf2 -local
If ${found_trim} || ${found_keyf} || ${found_keyf2}  # 直接カット位置設定
  If ${found_trim} {      # avs-Trim
    Set type_trim_in 2
    Set use_trim_keepcut 1  # カット位置は直接Amatsukazeが読み込むので補正なし
  }ElsIf ${found_keyf2} { # keyframe（chapter短時間構成カット）
    Set type_trim_in 3
    Set use_divforce     ${anysel_keyf2_divforce}  # 構成時間強制ファイル分割
    Set use_trim_keepcut ${anysel_keyf_keepcut}    # 内側カット補正無効設定
  }ElsIf ${found_keyf} {  # keyframe（カット位置情報）
    Set type_trim_in 3
    Set use_trim_keepcut ${anysel_keyf_keepcut}    # 内側カット補正無効設定
  };
  ReadCheck "${file_div}" -RegOut use_divfile  # 分割ファイル使用選択
  Set use_trim_direct 1    # 直接Trim位置を指定する
EndIf

#--- 直接trim位置設定時のCall処理追加 ---
Default jlsi_jltrim_run ""             # 直接位置設定用CallLスクリプト
Default jlsi_echo_run "${jlsi_jlrun}"  # Echo表示用選択スクリプト
If ${use_trim_direct}>0 {
  LocalSet cmt "直接位置指定"
  If ${use_divfile} { Set cmt "${cmt} （分割指定ファイルあり）" };
  If ${use_autodet} {                  # 直接位置設定後に構成推測する場合
    LocalSet cspace ""
    If "${jlsi_comment}"!="" {
      Set cspace "  + "
    };
    Set cmt "${cmt}${cspace}${jlsi_comment}"
    Set jlsi_echo_run "${jlsi_jlrun}  trim: "
  }else{
    Set jlsi_jlrun   ""                # 直接位置設定後は何も実行しない
    Set jlsi_echo_run ""
  };
  Set jlsi_jltrim_run "${file_calltr}"
  Set jlsi_comment "${cmt}"
  Set jlsi_echo_run "${jlsi_echo_run}${jlsi_jltrim_run}"
};
#--- Amatsukazeが直接入力ファイルの位置を読む時は内側にカット補正しない ---
If ${use_trim_keepcut}
  If outrev_cutframe { Set outrev_cutframe 0 };
EndIf

Echo 検索対象 : ${file_title}
Echo 放送局   : ${service_abbr} : ${service_id}
Echo 選択コメント : ${jlsi_comment}
Echo 選択スクリプト : ${jlsi_echo_run}

#--- 追加オプション設定 ---
If "${jlsi_optstr:Xq}"!="" {   # 変数内ダブルクォートは通常文字として認識させる
  Echo 追加オプション : ${jlsi_optstr}
  OptSet ${jlsi_optstr}
};

#--- 直接位置設定Call動作 ---
If "${jlsi_jltrim_run}"!="" {
  Call "${jlsi_jltrim_run}"
};
#--- Call動作 ---
If "${jlsi_jlrun}"!="" {
  Call "${jlsi_jlrun}"
};
